Verificador de Paridade

            
function verificarParidade(n) {
    let binario = converteParaBinario(n);
    let conta = contarUns(binario);

    if (verificaParidadeContagem(conta)) {
        return "Ímpar";
    } else {
        return "Par";
    }
}
                
function converteParaBinario(n) {
    return n.toString(2);
}
                
function contarUns(binario) {
    if (binario.length === 0) {
        return 0;
    } else {
        return (binario.charAt(0) === '1' ? 1 : 0) + contarUns(binario.substring(1));
    }
}
                
function verificaParidadeContagem(conta) {
    return conta % 2 === 0;
}
                
let numero = parseInt(prompt("Digite um número para verificar sua paridade: "), 10);
let resultado = verificarParidade(numero);
alert(resultado);
