Verificador de Paridade

                
                    function verificarParidade(n) {
                        let binario = n.toString(2);
                    
                        function contarUns(binario) {
                            if (binario.length === 0) {
                                return 0;
                            } else {
                                return (binario[0] === '1' ? 1 : 0) + contarUns(binario.slice(1));
                            }
                        }
                    
                        let paridade = contarUns(binario) % 2;
                        if (paridade === 0) {
                            return "Ímpar";
                        } else {
                            return "Par";
                        }
                    }
